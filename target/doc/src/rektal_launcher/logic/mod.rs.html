1/// The central logic module for R.E.K.T.A.L. launcher
2use std::process::Command;
3use std::sync::Mutex;
4use lazy_static::lazy_static;
5
6use crate::roles::Role;
7use crate::state::State;
8
9///Module internal state managed with lazy_static and Mutex
10lazy_static! {
11    static ref STATE: Mutex<State> = Mutex::new(State {
12        name: String::new(),
13        role: Role::Operator,
14        start_local: false,
15    });
16}
17
18/// Start the R.E.K.T.A.L. application with the current state
19pub(crate) fn start_rektal() -> std::io::Result<()> {
20    // Lock auf den State holen
21    let state = STATE.lock().unwrap();
22    println!("Starting R.E.K.T.A.L. as role: {}", state.role.as_str());
23
24    // TODO: Hier könnt Ihr args aus state oder anderen Variablen an Command übergeben
25    let status = Command::new("echo")
26        //.args(...)
27        .status()?;
28
29    println!("R.E.K.T.A.L. Exit-Code: {}", status);
30    Ok(())
31}
32
33/// Save the selected role into the module-internal state
34pub(crate) fn save_role(role: Role) {
35    let mut state = STATE.lock().unwrap();
36    state.role = role;
37    println!("Role saved: {}", state.role.as_str());
38}
39
40/// Save the entered name into the module-internal state
41pub(crate) fn save_name(name: String) {
42    let mut state = STATE.lock().unwrap();
43    state.name = name;
44    println!("Name saved: {}", state.name);
45}
46
47/// Set whether to start R.E.K.T.A.L. locally or not
48pub(crate) fn set_start_local(start: bool) {
49    let mut state = STATE.lock().unwrap();
50    state.start_local = start;
51    println!("Start local: {}", state.start_local);
52}
